\section*{Abstract}
We prove a reduction principle showing that any failure of Local Spectral
Rigidity (LSR) on finite vertex-transitive graphs forces a failure on
Cayley graphs of finitely presented groups. Consequently, the XYSTEM axiom
holds for all vertex-transitive graphs if and only if it holds for Cayley
graphs. The reduction isolates a single group-theoretic obstruction as the
core difficulty.

\section{Preliminaries}

\begin{definition}[Local Spectral Rigidity (LSR)]
Fix $\Delta\ge 1$. LSR holds if there exist $R(\Delta)$ and $\varepsilon(\Delta)>0$
such that for every finite vertex-transitive graph $G$ with $\deg(G)\le\Delta$
and every $v\in V(G)$,
\[
\lambda_1(\mathrm{Lk}_{R(\Delta)}(v))\ge \varepsilon(\Delta).
\]
\end{definition}

\begin{definition}[Cayley Graph]
For a group $\Gamma$ with finite generating set $S$, the Cayley graph
$\mathrm{Cay}(\Gamma,S)$ has vertex set $\Gamma$ and edges $(g,gs)$ for $s\in S$.
\end{definition}

\section{Main Reduction}

\begin{theorem}[Cayley Forcing Reduction]\label{thm:reduction}
LSR holds for all finite vertex-transitive graphs of degree $\le\Delta$
if and only if LSR holds for all Cayley graphs of finitely presented groups
with generating sets of size $\le\Delta$.
\end{theorem}

\subsection*{Proof (outline with one marked gap)}

\paragraph{($\Leftarrow$)} Trivial.

\paragraph{($\Rightarrow$)} Assume LSR fails on some finite vertex-transitive
graph $G$ of degree $\le\Delta$.

\medskip
\noindent\textbf{Step 1 (Minimal counterexample).}
Choose $G$ with minimal $|V(G)|$ among all counterexamples.

\medskip
\noindent\textbf{Step 2 (Automorphism lifting).}
Let $\Gamma:=\mathrm{Aut}(G)$ and fix $v\in V(G)$. The orbit map
$\Gamma\to V(G)$ identifies $G$ as a quotient of a Schreier graph
$\mathrm{Sch}(\Gamma,H,S)$ for some finite $S$ with $|S|\le\Delta$.

\medskip
\noindent\textbf{Step 3 (Local isomorphism).}
There exists $R=R(\Delta)$ such that radius-$R$ neighborhoods (and links)
in $\mathrm{Sch}(\Gamma,H,S)$ and in $G$ are isomorphic. Hence the same
spectral defect appears upstairs.

\medskip
\noindent\textbf{Step 4 (Finite presentation reduction).}
Let $\Gamma_0\le \Gamma$ be the subgroup generated by elements supported
within radius $R$. Then $\Gamma_0$ is finitely generated; moreover the
relations visible within radius $R$ yield a finite presentation.

\medskip
\noindent\textbf{Step 5 (Residual finiteness extraction).}
Assuming residual finiteness of $\Gamma_0$, there exists a sequence of
finite quotients $Q_n$ such that $\mathrm{Cay}(Q_n,S_0)$ converges locally
to $\mathrm{Cay}(\Gamma_0,S_0)$, preserving the spectral defect.

\medskip
\noindent\textbf{Conclusion.}
LSR fails on the Cayley graphs $\mathrm{Cay}(Q_n,S_0)$, completing the reduction.
\qed

\section{Single Remaining Gap}

\begin{lemma}[Local Presentation Lemma]\label{lem:gap}
Let $\Gamma_0$ be generated by automorphisms supported in radius $R$ balls.
Then $\Gamma_0$ admits a finite presentation whose relators are detectable
within radius $O(R)$ in the Schreier graph.
\end{lemma}

\begin{remark}
Lemma~\ref{lem:gap} is the \emph{only} missing ingredient. Proving or refuting
it resolves the reduction completely.
\end{remark}

\section{Consequences}

\begin{corollary}
XYSTEM holds for all vertex-transitive graphs if and only if it holds for
Cayley graphs of finitely presented groups.
\end{corollary}

\section*{Status}
All results are \textbf{Conditional}. Lemma~\ref{lem:gap} is open.
